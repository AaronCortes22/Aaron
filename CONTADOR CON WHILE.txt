CONTADOR CON WHILE
//contador del 1 al 100//

main ()
{
    i = 1;
    x = i; 

    DO
    {
        printf(x);
        x++
    } WHILE (x<101) //mientras que lleguemos al 101// 
    printf ("adios")
}
